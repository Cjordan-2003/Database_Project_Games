<script>
  // PHP generates this data at page load
  const gameData = {
    greatest: <?php echo getGamesJSON("rating DESC"); ?>,
    worst: <?php echo getGamesJSON("rating ASC"); ?>,
    oldest: <?php echo getGamesJSON("release_date ASC"); ?>,
    newest: <?php echo getGamesJSON("release_date DESC"); ?>
  };

  function showData(category) {
    const games = gameData[category];
    if (!games || games.length === 0) {
      document.getElementById('outputBox').innerHTML = "No data found.";
      return;
    }

    let html = "<ul>";
    games.forEach(g => {
      html += `<li>${g.title}</li>`;
    });
    html += "</ul>";

    document.getElementById('outputBox').innerHTML = html;
  }
</script>
